    <script>

        let figmaData = null;


        // Comprobar si ya hay credenciales guardadas al cargar la página

        window.onload = function() {

            const token = sessionStorage.getItem('figma_token');

            const projectId = sessionStorage.getItem('figma_project_id');

            if (token && projectId) {

                document.getElementById('login-screen').classList.add('hidden');

                document.getElementById('dashboard-screen').classList.remove('hidden');

                fetchFigmaData(token, projectId);

            }

        }


        function saveConfig() {

            const token = document.getElementById('figma-token').value.trim();

            const projectId = document.getElementById('project-id').value.trim();


            if (!token || !projectId) {

                alert('Por favor, rellena ambos campos.');

                return;

            }


            sessionStorage.setItem('figma_token', token);

            sessionStorage.setItem('figma_project_id', projectId);


            document.getElementById('login-screen').classList.add('hidden');

            document.getElementById('dashboard-screen').classList.remove('hidden');

            fetchFigmaData(token, projectId);

        }


        function disconnect() {

            sessionStorage.clear();

            window.location.reload();

        }


        // Petición real en tiempo real a la API de Figma

        async function fetchFigmaData(token, projectId) {

            try {

                const response = await fetch(`https://api.figma.com/v1/projects/${projectId}/files?branch_data=true`, {

                    headers: { 'X-Figma-Token': token }

                });


                if (!response.ok) throw new Error('Error al conectar con Figma. Verifica el token o el ID del proyecto.');


                figmaData = await response.json();

                

                // Rellenar el filtro de archivos dinámicamente

                const filterSelect = document.getElementById('file-filter');

                document.getElementById('project-title-tag').innerText = figmaData.name || 'Proyecto';

                

                figmaData.files.forEach(file => {

                    const option = document.createElement('option');

                    option.value = file.key;

                    option.innerText = file.name;

                    filterSelect.appendChild(option);

                });


                renderDashboard();

            } catch (error) {

                alert(error.message);

                disconnect();

            }

        }


        function renderDashboard() {

            if (!figmaData) return;


            const searchTerm = document.getElementById('search-input').value.toLowerCase();

            const fileFilter = document.getElementById('file-filter').value;

            const container = document.getElementById('branches-container');

            container.innerHTML = '';


            let totalFiles = figmaData.files.length;

            let totalBranches = 0;

            let staleBranches = 0;

            let branchesToRender = [];


            const ahora = new Date();


            figmaData.files.forEach(file => {

                // Filtrar por archivo si corresponde

                if (fileFilter !== 'all' && file.key !== fileFilter) return;


                if (file.branches && file.branches.length > 0) {

                    file.branches.forEach(branch => {

                        totalBranches++;

                        

                        // Calcular si es obsoleta (más de 90 días sin modificar)

                        const fechaMod = new Date(branch.last_modified);

                        const diasInactiva = (ahora - fechaMod) / (1000 * 60 * 60 * 24);

                        const isStale = diasInactiva > 90;

                        

                        if (isStale) staleBranches++;


                        // Filtrar por buscador de texto

                        if (branch.name.toLowerCase().includes(searchTerm)) {

                            branchesToRender.push({

                                ...branch,

                                originFileName: file.name,

                                isStale: isStale

                            });

                        }

                    });

                }

            });


            // Actualizar KPIs en la interfaz

            document.getElementById('kpi-files').innerText = totalFiles;

            document.getElementById('kpi-branches').innerText = totalBranches;

            document.getElementById('kpi-stale').innerText = staleBranches;


            if (branchesToRender.length === 0) {

                container.innerHTML = `<div class="px-6 py-8 text-center text-slate-500 text-sm">No se encontraron ramas abiertas que coincidan.</div>`;

                return;

            }


            // Renderizar la lista ordenando por fecha de modificación más reciente

            branchesToRender.sort((a, b) => new Date(b.last_modified) - new Date(a.last_modified));


            branchesToRender.forEach(branch => {

                const fecha = new Date(branch.last_modified).toLocaleDateString('es-ES', {

                    day: '2-digit', month: '2-digit', year: 'numeric', hour: '2-digit', minute:'2-digit'

                });


                const row = document.createElement('div');

                row.className = `flex flex-col sm:flex-row sm:items-center justify-between px-6 py-4 hover:bg-slate-700/30 transition ${branch.isStale ? 'bg-rose-950/10 border-l-2 border-rose-500' : ''}`;

                

                row.innerHTML = `

                    <div class="flex-1 min-w-0 pr-4">

                        <div class="flex items-center space-x-2">

                            <span class="text-sm font-semibold text-white truncate">${branch.name}</span>

                            ${branch.isStale ? '<span class="bg-rose-500/10 border border-rose-500/30 text-rose-400 text-[10px] px-2 py-0.5 rounded font-medium tracking-wide">OBSOLETA</span>' : ''}

                        </div>

                    </div>

                    <div class="w-full sm:w-48 mt-1 sm:mt-0">

                        <span class="inline-block bg-slate-900 border border-slate-700 text-slate-400 text-xs px-2.5 py-1 rounded font-medium">${branch.originFileName}</span>

                    </div>

                    <div class="w-full sm:w-48 mt-1 sm:mt-0 text-xs text-slate-400">

                        ${fecha}

                    </div>

                    <div class="mt-2 sm:mt-0 text-right">

                        <a href="https://www.figma.com/file/${branch.key}" target="_blank" class="inline-flex items-center text-xs font-medium bg-slate-700 hover:bg-orange-600 text-white px-3 py-1.5 rounded transition shadow-sm">

                            Abrir Rama

                        </a>

                    </div>

                `;

                container.appendChild(row);

            });

        }

    </script>
